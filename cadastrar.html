function cadastrar() { const usuario = document.getElementById("usuario").value.trim(); const senha = document.getElementById("senha").value.trim(); const confirmar = document.getElementById("confirmar").value.trim(); const nome = document.getElementById("nome").value.trim(); const btn = document.querySelector(".btn"); if (!usuario || !senha || !confirmar || !nome) { mostrarMensagem("❌ Preencha todos os campos", "erro"); return; } if (senha !== confirmar) { mostrarMensagem("❌ As senhas não coincidem", "erro"); return; } if (senha.length < 4) { mostrarMensagem("❌ A senha deve ter pelo menos 4 caracteres", "erro"); return; } btn.disabled = true; btn.innerHTML = ' Cadastrando...'; const callbackName = "jsonpCadastro_" + Date.now(); window[callbackName] = function(r) { delete window[callbackName]; document.body.removeChild(script); btn.disabled = false; btn.innerHTML = "CADASTRAR"; if (r.sucesso) { mostrarMensagem("✅ " + r.mensagem, "ok"); setTimeout(() => { window.location.href = "login.html"; }, 1500); } else { mostrarMensagem("❌ " + r.mensagem, "erro"); } }; const script = document.createElement('script'); script.src = `${API_URL}?acao=cadastrar&usuario=${encodeURIComponent(usuario)}&senha=${encodeURIComponent(senha)}&nome=${encodeURIComponent(nome)}&callback=${callbackName}`; script.onerror = function() { btn.disabled = false; btn.innerHTML = "CADASTRAR"; mostrarMensagem("❌ Erro no servidor", "erro"); delete window[callbackName]; document.body.removeChild(script); }; document.body.appendChild(script); }
