function login() { const usuario = document.getElementById("usuario").value.trim(); const senha = document.getElementById("senha").value; const btn = document.getElementById("btnLogin"); const msg = document.getElementById("msg"); msg.innerHTML = ""; msg.className = "msg"; if (!usuario || !senha) { msg.classList.add("erro"); msg.innerHTML = "❌ Preencha usuário e senha"; return; } btn.disabled = true; btn.innerHTML = ' Entrando...'; // Usar JSONP para evitar CORS const callbackName = "jsonpCallback_" + Date.now(); window[callbackName] = function(r) { delete window[callbackName]; document.body.removeChild(script); btn.disabled = false; btn.innerHTML = "ENTRAR NO SISTEMA"; if (r.sucesso) { localStorage.setItem("rupe_token", r.token); localStorage.setItem("rupe_nome", r.nome); localStorage.setItem("rupe_usuario", r.usuario); localStorage.setItem("rupe_nivel", r.nivel); msg.classList.add("ok"); msg.innerHTML = `👋 Bem-vindo, ${r.nome || r.usuario}!`; setTimeout(() => { window.location.href = "home.html"; }, 1000); } else { msg.classList.add("erro"); msg.innerHTML = "❌ " + (r.mensagem || r.erro || "Login inválido"); } }; const script = document.createElement('script'); script.src = `${API_URL}?acao=login&usuario=${encodeURIComponent(usuario)}&senha=${encodeURIComponent(senha)}&callback=${callbackName}`; script.onerror = function() { btn.disabled = false; btn.innerHTML = "ENTRAR NO SISTEMA"; msg.classList.add("erro"); msg.innerHTML = "❌ Erro no servidor"; delete window[callbackName]; document.body.removeChild(script); }; document.body.appendChild(script); }
