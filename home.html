function carregarStats() { const callbackName = "jsonpStats_" + Date.now(); window[callbackName] = function(r) { delete window[callbackName]; document.body.removeChild(script); if (r && r.resumo) { document.getElementById("total").innerHTML = r.resumo.total || 0; document.getElementById("rupe").innerHTML = r.resumo.rupe || 0; document.getElementById("relatoria").innerHTML = r.resumo.relatoria || 0; } else if (r && r.erro) { console.error("Erro da API:", r.erro); } }; const script = document.createElement('script'); script.src = `${API_URL}?acao=dashboard&callback=${callbackName}`; script.onerror = function() { console.error("Erro ao carregar stats"); delete window[callbackName]; document.body.removeChild(script); }; document.body.appendChild(script); }
